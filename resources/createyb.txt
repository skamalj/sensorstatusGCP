CREATE TABLE raw_data(
device_id INT,
sensor_type VARCHAR,
factory_id INT,
section VARCHAR,
value INT,
eventtime BIGINT,
PRIMARY KEY(device_id,eventtime))
WITH CLUSTERING ORDER BY (eventtime DESC);

CREATE TABLE status (
    device_id INT,
    eventtime TIMESTAMP,
    updated_at TIMESTAMP,
    sensor_type TEXT,
    factory_id INT,
    section TEXT,
    status TEXT,
    PRIMARY KEY(device_id)
)
